A.10 Unlock Files

Code
unlocked_files <- character()

for (i in files) {
  # if (!file_exists(i)) next

  i_path <-
    i |>
    unlock_file(
      private_key = private_key,
      suffix = ".lockr",
      remove_file = TRUE,
      password = password
    )

  cat_line()

  unlocked_files <- c(unlocked_files, i_path)
}
